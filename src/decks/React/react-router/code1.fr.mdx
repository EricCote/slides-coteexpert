```jsx src/Menu2.jsx
const menuList = [
  {
    name: 'Home',
    url: '/',
  },
  {
    name: 'News',
    url: '/news',
  },
  {
    name: 'Contact',
    url: '/contact',
  },
  {
    name: 'About',
    url: '/about',
  },
];

export default function Menu() {
  return (
    <ul className='menu'>
      {menuList.map((menuItem) => (
        <li key={menuItem.url}>
          <a
            className={menuItem.url === '/' ? 'active' : ''}
            href={menuItem.url}
          >
            {menuItem.name}
          </a>
        </li>
      ))}
    </ul>
  );
}
```

```jsx src/Home.jsx
export default function Home() {
  return (
    <div>
      <h1>Home page</h1>
      <p>Règles d'un serveur web en mode SPA:</p>
      <ul>
        <li>
          Si un dossier ou fichier existe sur le disque, on retourne le contenu.
        </li>
        <li>Si ça n'existe pas, on sert la page par défaut (index.html)</li>
        <ul>
          <li>
            À moins qu'il n'y ait une extension (.png .css .js .html) on sert
            une erreur 404.
          </li>
        </ul>
        <li>Différents Frameworks SPA ont différentes règles</li>
      </ul>
    </div>
  );
}
```
