```jsx src/Menu2.jsx
const menuList = [
  {
    name: 'Home',
    url: '/',
  },
  {
    name: 'News',
    url: '/news',
  },
  {
    name: 'Contact',
    url: '/contact',
  },
  {
    name: 'About',
    url: '/about',
  },
];

export default function Menu() {
  return (
    <ul className='menu'>
      {menuList.map((menuItem) => (
        <li key={menuItem.url}>
          <a
            className={menuItem.url === '/' ? 'active' : ''}
            href={menuItem.url}
          >
            {menuItem.name}
          </a>
        </li>
      ))}
    </ul>
  );
}
```

```jsx src/Home.jsx
export default function Home() {
  return (
    <div>
      <h1>Home page</h1>
      <p>The way a web server is configured in SPA mode:</p>
      <ul>
        <li>If a folder or page exists on disk, serve it.</li>
        <li>
          If it doesn't exist, serve the default page instead (index.html)
        </li>
        <ul>
          <li>
            Unless there is an extension in the request (.png .css .js .html)
            then serve a 404.
          </li>
        </ul>
        <li>Different SPA frameworks might have slightly different rules</li>
      </ul>
    </div>
  );
}
```
